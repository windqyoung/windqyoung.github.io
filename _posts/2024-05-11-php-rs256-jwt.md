function base64_url_json_encode($data, $json = true)
{
    $data = $json ? json_encode($data) : $data;
    return str_replace(['+', '/', '='], ['-', '_', ''], base64_encode($data));
}

function base64_url_json_decode($data, $json = true)
{
    $data = str_replace(['-', '_'], ['+', '/'], $data);
    $data .= str_repeat('=', 4 - strlen($data) % 4);
    $bd = base64_decode($data);
    return $json ? json_decode($bd, true) : $bd;
}
